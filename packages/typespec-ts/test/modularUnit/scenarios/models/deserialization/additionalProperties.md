    ...Record<string>;
    propA: string;
    propB: string;
}

@route("/serialize")

}

```yaml

```

    propB: string;

    propB: string;

  @route("bar")
  op bar(): { @body body: SimpleModel };

  op baz(): { @body body: EmptyModel };
  @route("bas")
  op bas(): { @body body: UnionModel };
  @route("bab")
  op bab(): { @body body: NameConflictModel };
  @route("obj")
